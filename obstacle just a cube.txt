export function getObstacle (width, height, depth, veinTex) {
    veinTex.wrapS = THREE.RepeatWrapping;
    veinTex.wrapT = THREE.RepeatWrapping;
    var tex=veinTex.clone();
    tex.needsUpdate = true;
    tex.repeat.set(depth/height*2,height/height*2);

    const material_base = new THREE.MeshPhongMaterial( { color: 0x800000, map: tex, bumpMap: tex } );//112 164
    const geometry_1 = new THREE.BoxGeometry(width, height, depth);

    const obstacle = new THREE.Mesh( geometry_1, material_base);
    obstacle.position.set(0.0, height/2, 0.0);
    obstacle.translateY(-25);
    return(obstacle);
}
